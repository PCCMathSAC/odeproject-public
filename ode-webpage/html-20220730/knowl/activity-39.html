(a)
\begin{align*} \frac{dx}{dt} & = x(2 + x) + 2xy\\ \frac{dy}{dt} & = y(2 + y) + 2xy \end{align*}
(b)
\begin{align*} \frac{dx}{dt} & = x(2 + x) -2xy\\ \frac{dy}{dt} & = y(2 + y) - 2xy \end{align*}
(c)
\begin{align*} \frac{dx}{dt} & = x(2 + x) + 2xy\\ \frac{dy}{dt} & = y(2 + y) - 2xy \end{align*}
(d)
\begin{align*} \frac{dx}{dt} & = x(2 + x) + 2xy\\ \frac{dy}{dt} & = y^2(2 + y) - 2xy \end{align*}
